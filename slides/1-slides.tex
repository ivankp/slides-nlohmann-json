\begin{frame}[fragile]\frametitle{JSON}
  \begin{itemize}
    \item JavaScript Object Notation
    \item Data types:
      null, boolean, number, string, array, object
    \item {\bfseries Tree structure}: branches are either objects or arrays
      \begin{itemize}
        \item Object keys are strings
        \item Array keys are indices
      \end{itemize}
    \item No comments
  \end{itemize}
\begin{lstlisting}[language=json]
{
    "key": [
        null, true, false, 42, 1.41, "string"
    ]
}
\end{lstlisting}
\end{frame}

\begin{frame}[fragile]
\frametitle{\texorpdfstring{\ttfamily nlohmann/json}{nlohmann/json}}
  \begin{itemize}
    % \setlength\itemsep{1.5em}
    \item {\footnotesize\url{https://github.com/nlohmann/json}}
    \item {\footnotesize\url{https://json.nlohmann.me/api/basic_json/}}
    \item Development started in 2013
    \item Header-only library
    \item By default, uses standard library containers to represent JSON values
    \item Tries to mimic access patterns of the standard library
  \end{itemize}
\end{frame}

\begin{frame}[fragile]\frametitle{Parsing}
  \begin{itemize}
    \item \lstinline{@Tbasic_json@     @Tbasic_json@::@Fparse@();} \\
      {\scriptsize\url{https://json.nlohmann.me/api/basic_json/parse/}}
      \begin{itemize}
        \item \lstinline{auto j = @Tjson@::@Fparse@(std::@Tifstream@("/opt/data.json"));}
        \item \lstinline{auto j = @Tjson@::@Fparse@("[ 1, 2, null ]");}
      \end{itemize}
    \item \lstinline{std::@Tistream@& operator@F>>@(std::@Tistream@&, @Tbasic_json@&);} \\
      {\scriptsize\url{https://json.nlohmann.me/api/operator_gtgt/}}
      \vspace{-6pt}
      \begin{itemize}
        \item \begin{lstlisting}
@Tjson@ j;
std::@Tifstream@("/opt/data.json") >> j;
        \end{lstlisting}
      \end{itemize}
      \vspace{-8pt}
    \item \lstinline{bool @Tbasic_json@::@Faccept@();} \\
      {\scriptsize\url{https://json.nlohmann.me/api/basic_json/accept/}}
      \begin{itemize}
        \item Accepts same types as \lstinline{@Fparse@()}
        \item Only checks if the input is valid JSON
      \end{itemize}
    \item \lstinline{@Tjson@   operator @L""_json@(const char*, std::size_t);} \\
      {\scriptsize\url{https://json.nlohmann.me/api/operator_literal_json/}}
      \begin{itemize}
        \item \lstinline{auto j = "[ 1, 2, null ]"@L_json@;}
      \end{itemize}
  \end{itemize}

\begin{frame}[fragile]\frametitle{Exceptions}
  \begin{itemize}
    \item {\footnotesize\url{https://json.nlohmann.me/api/basic_json/exception/}}
    \item \lstinline{@Nnlohmann@::@Tbasic_json@::@Texception@;}
    \item \lstinline{class @Texception@ : public std::@Texception@;}
    \item Subclasses:
    \begin{itemize}
      \item \lstinline{@Tparse_error@}
        for exceptions indicating a parse error
      \item \lstinline{@Tinvalid_iterator@}
        for exceptions indicating errors with iterators
      \item \lstinline{@Ttype_error@}
        for exceptions indicating executing a member function with a wrong type
      \item \lstinline{@Tout_of_range@}
        for exceptions indicating access out of the defined range
      \item \lstinline{@Tother_error@}
        for exceptions indicating other library errors
    \end{itemize}
  \end{itemize}
