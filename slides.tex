\slidetitle{JSON}

\begin{itemize}
\item JavaScript Object Notation
\item Data types:
  null, boolean, number, string, array, object
\item {\bfseries Tree structure}: branches are either objects or arrays
  \begin{itemize}
    \item Object keys are strings
    \item Array keys are indices
  \end{itemize}
\item No comments
\end{itemize}

\vspace{1em}
\begin{lstlisting}[language=json]
{
    "key": [
        null, true, false, 42, 1.41, "string"
    ]
}
\end{lstlisting}

\slidetitle{\ttfamily nlohmann/json}

\begin{itemize}
  \item \faFileTextO\ \ \url{https://github.com/nlohmann/json}
  \item \faGithub\ \ \url{https://json.nlohmann.me/api/basic_json/}
  \item Developed since 2013
  \item Header-only library
  \item Uses standard library containers to represent JSON values
  \begin{itemize}
    \item This is configurable via \lstinline{basic_json} template parameters
  \end{itemize}
  \item Tries to mimic access patterns of the standard library
\end{itemize}

\slidetitle{Parsing}

\begin{itemize}
  \item \begin{lstlisting}[basicstyle={\ttfamily\fontsize{18pt}{18pt}\selectfont}]
@Tbasic_json@ @Tbasic_json@::@Fparse@(); @!\hfill\text{%
      \href{https://json.nlohmann.me/api/basic_json/parse/}{\faFileTextO}}@
    \end{lstlisting}
    \begin{itemize}
      \item \begin{lstlisting}
auto j = @Tjson@::@Fparse@(std::@Tifstream@("/opt/data.json"));
      \end{lstlisting}

      \item \begin{lstlisting}
auto j = @Tjson@::@Fparse@("[ 1, 2, null ]");
      \end{lstlisting}
    \end{itemize}

  \item \begin{lstlisting}[basicstyle={\ttfamily\fontsize{18pt}{18pt}\selectfont}]
std::@Tistream@& operator>>(std::@Tistream@&, @Tbasic_json@&); @!\hfill\text{%
      \href{https://json.nlohmann.me/api/operator_gtgt/}{\faFileTextO}}@
    \end{lstlisting}
    \begin{itemize}
      \item \begin{lstlisting}
@Tjson@ j;
std::@Tifstream@("/opt/data.json") >> j;
      \end{lstlisting}
    \end{itemize}

  \item \begin{lstlisting}[basicstyle={\ttfamily\fontsize{18pt}{18pt}\selectfont}]
bool @Tbasic_json@::@Faccept@(); @!\hfill\text{%

    \end{lstlisting}
    \begin{itemize}
      \item Accepts same types as \lstinline{@Fparse@()}
      \item Only checks if the input is valid JSON
